\begin{algorithm}[t]
	\caption{Calculate $y = x^n$}
	\label{alg:algorithm}
	\begin{algorithmic}[1]
		\Require	$n \geq 0 \vee x \neq 0$
		\Ensure	$y = x^n$
		\State $y \Leftarrow 1$
			\If{$n < 0$}
				\State $X \Leftarrow 1 / x$
				\State $N \Leftarrow -n$
			\Else
				\State $X \Leftarrow x$
				\State $N \Leftarrow n$
			\EndIf
		\While{$N \neq 0$}
			\If{$N$ is even}
				\State $X \Leftarrow X \times X$
				\State $N \Leftarrow N / 2$
			\Else[$N$ is odd]
				\State $y \Leftarrow y \times X$
				\State $N \Leftarrow N - 1$
			\EndIf
		\EndWhile
	\end{algorithmic}
\end{algorithm}
